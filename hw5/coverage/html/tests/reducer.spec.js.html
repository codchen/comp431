import { expect } from 'chai'
import { createStore } from 'redux'
 
import reducer from '../src/reducers'
import { successMessage, errorMessage, MessageTypes } from '../src/actions'
import { setArticles, searchByAuthor } from '../src/components/article/container/articleActions'
 
describe('Validate reducer', () => {
    const store = createStore(reducer)
 
    it('should initialize state', (done) => {
        expect(store.getState()).to.be.ok
        done()
    })
 
    it('should state success', (done) => {
        store.dispatch(successMessage('succ'))
        expect(store.getState().page.messageType).to.eql(MessageTypes.SUCCESS)
        done()
    })
 
    it('should state error', (done) => {
        store.dispatch(errorMessage('err'))
        expect(store.getState().page.messageType).to.eql(MessageTypes.FAILURE)
        done()
    })
 
    it('should set the articles', (done) => {
        const articles = [
            {
                _id: 0,
                author: 'Tony',
                comments: [],
                date: '',
                img: '',
                text: '123'
            }, {
                _id: 1,
                author: 'Seb',
                comments: [],
                date: '',
                img: '',
                text: '456'
            }
        ]
        store.dispatch(setArticles(articles))
        expect(store.getState().article.articles.length).to.eql(articles.length)
        expect(store.getState().article.articles[0]._id).to.eql(articles[0]._id)
        expect(store.getState().article.articles[1]._id).to.eql(articles[1]._id)
        expect(store.getState().article.articles[0].comments).to.eql(articles[0].comments)
        expect(store.getState().article.articles[1].comments).to.eql(articles[1].comments)
        expect(store.getState().article.articles[0].author).to.eql(articles[0].author)
        expect(store.getState().article.articles[1].author).to.eql(articles[1].author)
        expect(store.getState().article.articles[0].img).to.eql(articles[0].img)
        expect(store.getState().article.articles[1].img).to.eql(articles[1].img)
        expect(store.getState().article.articles[0].text).to.eql(articles[0].text)
        expect(store.getState().article.articles[1].text).to.eql(articles[1].text)
        done()
    })
 
    it('should set the search keyword', (done) => {
        const keyword = 'Tony'
        store.dispatch(searchByAuthor(keyword))
        expect(store.getState().article.keyword).to.eql(keyword)
        done()
    })
 
    it('should filter displayed articles by the search keyword', (done) => {
        expect(store.getState().article.visibleArticles.length).to.eql(1)
        expect(store.getState().article.visibleArticles[0].author).to.eql('Tony')
        done()
    })
})
