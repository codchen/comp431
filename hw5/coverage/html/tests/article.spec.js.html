import React from 'react'
import { expect } from 'chai'
import TestUtils from 'react-addons-test-utils'
import { findDOMNode } from 'react-dom'
 
import SearchArticle from '../src/components/article/container/presentational/searchArticle'
import * as actions from '../src/components/article/container/articleActions'
 
describe('Validate Article actions', () => {
    const articles = [
        {
            _id: 0,
            author: 'Tony',
            comments: [],
            date: '',
            img: '',
            text: '123'
        }, {
            _id: 1,
            author: 'Seb',
            comments: [],
            date: '',
            img: '',
            text: '456'
        }
    ]
    
    it('should update search keyword', (done) => {
    	let keyword = ''
    	const options = [
		    { text: 'All', callback: () => keyword = actions.showAll().keyword },
		    { text: 'By Text', callback: (text) => keyword = actions.searchByText(text).keyword },
		    { text: 'By Author', callback: (author) => keyword = actions.searchByAuthor(author).keyword }
		]
    	const node = TestUtils.renderIntoDocument((<div>
            <SearchArticle options={ options } />
        </div>))
 
        const form = findDOMNode(node).children[0]
        const tf = form.children[0].children[0].children[0]
        const select = form.children[0].children[1].children[0]
        const btn = form.children[0].children[2].children[0]
 
        tf.value = 'Tony'
        select.value = 'By Author'
        TestUtils.Simulate.change(select)
        TestUtils.Simulate.click(btn)
        expect(keyword).to.eql('Tony')
 
        tf.value = 'flower'
        select.value = 'By Text'
        TestUtils.Simulate.change(select)
        TestUtils.Simulate.click(btn)
        expect(keyword).to.eql('flower')
 
        select.value = 'All'
        TestUtils.Simulate.change(select)
        TestUtils.Simulate.click(btn)
        expect(keyword).to.eql('')
    	done()
    })
})
