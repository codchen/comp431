import { expect } from 'chai'
import { createStore, applyMiddleware } from 'redux'
import thunkMiddleware from 'redux-thunk'
 
import { url, resource } from '../src/utils/ajax'
import { page } from '../src/reducers'
import { Locations, toLanding, toMain, toProfile, updateErrorMessage, updateSuccessMessage, noop, clearMessage } from '../src/actions'
import fetch, { mock } from 'mock-fetch'
 
describe('Validate actions', () => {
	beforeEach(() => {
        global.fetch = fetch
    })
 
    it('resource should be a resource', (done) => {
    	const text = 'Got the resource'
    	const endpoint = 'test-resource'
    	mock(`${url}/${endpoint}`, {
    		method: 'GET',
    		headers: { 'Content-Type': 'text/plain' },
    		text: text
    	})
    	resource('GET', endpoint, '')
    		.then(r => {
    			expect(r).to.eql(text)
    		})
    		.then(done)
    		.catch(done)
    })
 
    it('resource should give me the http error', (done) => {
    	const statusText = 'Forbidden'
    	const endpoint = 'test-error'
    	mock(`${url}/${endpoint}`, {
    		method: 'GET',
    		status: 403,
    		statusText: statusText,
    		headers: { 'Content-Type': 'text/plain' },
    		text: 'not relevant'
    	})
    	resource('GET', endpoint, '')
    		.catch(reason => {
    			expect(reason.message).to.contain(statusText)
    		})
    		.then(done)
    		.catch(done)
    })
 
    it('resource should be POSTable', (done) => {
    	const text = 'POSTed'
    	const payload = {
    		dummy: 'dummy'
    	}
    	const endpoint = 'test-post'
    	mock(`${url}/${endpoint}`, {
    		method: 'POST',
    		headers: { 'Content-Type': 'text/plain' },
    		text: text
    	})
    	resource('POST', endpoint, payload)
    		.then(r => {
    			expect(r).to.eql(text)
    		})
    		.then(done)
    		.catch(done)
    })
 
    const store = createStore(page, applyMiddleware(thunkMiddleware))
 
    it('should update error message', (done) => {
        const errMsg = 'error message'
        store.dispatch(updateErrorMessage(errMsg))
        expect(store.getState().messageText).to.eql(errMsg)
        done()
    })
 
    it('should update success message', (done) => {
        const succMsg = 'success message'
        store.dispatch(updateSuccessMessage(succMsg))
        expect(store.getState().messageText).to.eql(succMsg)
        done()
    })
 
    it('should navigate to profile', (done) => {
        store.dispatch(toProfile())
        expect(store.getState().location).to.eql(Locations.PROFILE_PAGE)
        done()
    })
 
    it('should navigate to main', (done) => {
        store.dispatch(toMain())
        expect(store.getState().location).to.eql(Locations.MAIN_PAGE)
        done()
    })
 
    it('should navigate to landing', (done) => {
        store.dispatch(toLanding())
        expect(store.getState().location).to.eql(Locations.LANDING_PAGE)
        done()
    })
 
    it('should clear message', (done) => {
        store.dispatch(clearMessage(0))
        setTimeout(() => {
            expect(store.getState().messageText).to.eql('')
            done()
        }, 10)
    })
 
    it('should do nothing on no-op', (done) => {
        const state = store.getState()
        store.dispatch(noop())
        expect(store.getState()).to.eql(state)
        done()
    })
})
