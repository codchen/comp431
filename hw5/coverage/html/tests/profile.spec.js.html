import { expect } from 'chai'
import { createStore, applyMiddleware } from 'redux'
import thunkMiddleware from 'redux-thunk'
 
import { url, resource } from '../src/utils/ajax'
import home from '../src/components/main/container/homeReducer'
import { fetchEmail, fetchZipcode, fetchDob } from '../src/components/profile/container/profileActions'
import { fetchHeadline, fetchAvatar, updateHeadline } from '../src/components/main/container/homeActions'
import fetch, { mock } from 'mock-fetch'
 
describe('Validate Profile actions', () => {
	beforeEach(() => {
        global.fetch = fetch
    })
 
    const store = createStore(home, applyMiddleware(thunkMiddleware))
    const username = 'xc12'
    const email = 'xc12@rice.edu'
    const zipcode = '77005'
    const dob = (new Date('02/14/1995')).getTime()
    const avatar = 'dummy'
    const headline = 'original headline'
 
    it('should fetch the user\'s proile information', (done) => {
    	mock(`${url}/email`, {
            method: 'GET',
            headers: { 'Content-Type': 'application/json' },
            json: {
                username,
                email
            }
        })
        mock(`${url}/zipcode`, {
            method: 'GET',
            headers: { 'Content-Type': 'application/json' },
            json: {
                username,
                zipcode
            }
        })
        mock(`${url}/dob`, {
            method: 'GET',
            headers: { 'Content-Type': 'application/json' },
            json: {
                username,
                dob
            }
        })
        mock(`${url}/headlines`, {
            method: 'GET',
            headers: { 'Content-Type': 'application/json' },
            json: {
                headlines: [
                    {
                        username,
                        headline
                    }
                ]
            }
        })
        mock(`${url}/avatars`, {
            method: 'GET',
            headers: { 'Content-Type': 'application/json' },
            json: {
                avatars: [
                    {
                        username,
                        avatar
                    }
                ]
            }
        })
 
        fetchEmail()
            .then((result) => {
                expect(result).to.eql(email)
            })
            .catch(done)
 
        fetchZipcode()
            .then((result) => {
                expect(result).to.eql(zipcode)
            })
            .catch(done)
 
        fetchDob()
            .then((result) => {
                expect(result).to.eql(dob)
            })
            .catch(done)
 
        fetchHeadline()
            .then((result) => {
                expect(result).to.eql(headline)
            })
            .catch(done)
 
        fetchAvatar()
            .then((result) => {
                expect(result).to.eql(avatar)
            })
            .then(done)
            .catch(done)
    })
 
    it('should update headline', (done) => {
        mock(`${url}/headline`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            json: {
                username,
                headline
            }
        })
 
        const newHeadline = 'new headline'
 
        store.dispatch(updateHeadline(newHeadline))
            .then((_) => {
                expect(store.getState().headline).to.eql(headline)
            })
            .then(done)
            .catch(done)
    })
})
